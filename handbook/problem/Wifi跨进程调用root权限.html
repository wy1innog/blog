    Wifi测试程序无root权限

    wyanlin

    # Wifi测试程序无root权限

    # 环境

    基于展锐Android11

    # 背景

    该平台支持双wifi，需要添加wifi0，wifi1测试项，wifi0的测试大概是控制系统api打开wifi功能，搜索附近的wifi并显示出来；

    而Wifi1的手动测试步骤如下：

    1. 在shell中输入两条指令用于加载驱动，此时ifconfig可以看到新出来eth1的网口

    2. 使用wpa_cli开源工具进行测试：

      /vendor/bin/hw/wpa_cli -i eth1 scan
/vendor/bin/hw/wpa_cli -i eth1 scan_result
      1
      2

      第一条指令用于开始搜索，一般会返回OK，多次发送会返回BUSY，第二条指令就是获取搜索到的wifi列表了

    将获取到的wifi信息显示于MAT程序上，这里只关注wifi1的测试流程如何用代码实现

    # 方法

    原本以为只需要在系统App中使用Runtime.getRuntime().exec执行获取返回结果就可以了，结果一直获取不到返回接口，最后发现是wpa_cli那里操作网卡的权限不够，系统应用只是拥有了system权限，但是操作网卡需要root权限，因此考虑通过跨进程通信，由MAT程序将需要做的事情告诉具有root权限的程序来运行，并将结果返回，这里使用了LocalSocket来通信

      wifitesthelper.c内容：

      #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <sys/un.h>
#include <stddef.h>
#include <log/log.h>
#include <cutils/sockets.h>
#include <pthread.h>
#include "wifiTestHelper.h"

#define _LINE_LENGTH 1024
#define PATH "wifihelper"
#define LOG_TAG "wifitesthelper"
bool isExit = false;

int load_driver(SOCKET_CMD *cmd_data) {
    system("insmod /mnt/vendor/socko/ssv6x5x.ko stacfgpath=/vendor/firmware/ssv6x5x-wifi.cfg");
    char command[100];
    sleep(2);
    sprintf(command, "/vendor/bin/hw/wpa_supplicant -Dnl80211 -i%s -c/vendor/firmware/wpa_supplicant.conf &",
     cmd_data->device_id == DEVICE_WLAN_0 ? WLAN_0 : WLAN_1);
    system(command);
	return 0;
}

int wifi_scan(SOCKET_CMD *cmd_data, char *reply, size_t size) {
    FILE* file = NULL;
    char command[80];

    sprintf(command, "/vendor/bin/hw/wpa_cli -i %s scan", cmd_data->device_id == DEVICE_WLAN_0 ? WLAN_0 : WLAN_1);
    if (NULL == (file = popen(command, "r"))) {
        ALOGD("%s execute command [scan] failed", __func__);
        return 0;
    }

    while (NULL != fgets(reply, size, file)) {
        if ((strlen(reply) > 1) && (reply[strlen(reply) - 1] == '\n')) {
           reply[strlen(reply) - 1] = '\0';
        }
    }
    pclose(file);
    ALOGD("%s result: %s", __func__, reply);
    return 1;
}

int wifi_scan_result(SOCKET_CMD *cmd_data, char *reply, size_t size) {
    FILE* file;
    char command[120];
    sprintf(command, "/vendor/bin/hw/wpa_cli -i %s scan_result", cmd_data->device_id == DEVICE_WLAN_0 ? WLAN_0 : WLAN_1);
    if (NULL == (file = popen(command, "r"))) {
        ALOGD("execute command [scan_result] failed");
        return -1;
    }

    fread(reply, size, 1, file);
    pclose(file);
    ALOGD("%s result: %s ", __func__, reply);
    return 1;
}

void * connectThread(void *arg)
{
    int ret;
    int socketID =*(int *)arg;
    if(socketID < 0){
        ALOGD("socketID is %d\n",socketID);
        return NULL;
    }
    char buff[2] = {0};
    char reply[_LINE_LENGTH];
    while (!isExit)
    {
        memset(buff, 0, sizeof(buff));
        ret = read(socketID, buff, sizeof(buff));
        if(ret < 0){
            ALOGD("receved failed\n");
            return NULL;
        }
        if (ret > 0) {
            ALOGD("WifiTestHelper server recived: cmd: %d deviceId: %d\n", buff[0], buff[1]);

            memset(reply, 0, sizeof(reply));
            SOCKET_CMD *cmd_data = (SOCKET_CMD *)malloc(sizeof(SOCKET_CMD));
            cmd_data->cmd = buff[0];
            cmd_data->device_id = buff[1];

            int need_reply = dispatchCmd(cmd_data, reply, sizeof(reply));
            if (need_reply) {
                ret = write(socketID, reply, strlen(reply));
                if(ret < 0){
                    ALOGD("write failed\n");
                    return NULL;
                }
            }
            free(cmd_data);
            cmd_data = NULL;
        }
    }
    ALOGD("connectThread exit connection");
    close(socketID);
    return NULL;
 
}

int dispatchCmd(SOCKET_CMD *cmd_data, char *reply, size_t size) {
    switch (cmd_data->cmd)
    {
    case CMD_LOAD_DRIVER:
        return load_driver(cmd_data);
    case CMD_SCAN:
        return wifi_scan(cmd_data, reply, size);
    case CMD_SCAN_RESULT:
        return wifi_scan_result(cmd_data, reply, size);
    default:
        isExit = true;
        return 0;
    }
}

int main(void){
    int ret ;
    int serverID = socket_local_server(PATH, ANDROID_SOCKET_NAMESPACE_ABSTRACT, SOCK_STREAM);
    if(serverID < 0){
        ALOGD("socket_local_server failed :%d\n",serverID);
        return serverID;
    }
    ALOGD("start server success!, listening ...\n");
    int socketID;
    pthread_t tid;
    while((socketID = accept(serverID, NULL, NULL)) >=0 ){
        ALOGD("accept socketId: %d\n", socketID);
        ret = pthread_create(&tid, NULL, connectThread,(void *)&socketID); 
        if(ret != 0) {
            ALOGD("error create thread:%s\n",strerror(ret));
            exit(1);
        }
    }
    return ret;
}

    2. MAT程序添加客户端，在需要测试wifi项的时候再与服务端连接，做相应的操作，连接和交互的部分代码：

      package com.sprd.validationtools.itemstest.wifi;

import android.net.LocalSocket;
import android.net.LocalSocketAddress;
import android.util.Log;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.io.PrintWriter;

/**
 * @Author: wyl
 * @CreateTime: 2023-09-07  14:14
 * @Description: TODO
 * @Version: 1.0
 */
public class ClientConnect {
    private static final String TAG = "ClientConnect";
    private static final String ADDRESS = "wifihelper";
    public static final byte CMD_LOAD_DRIVER = 0x01;
    public static final byte CMD_SCAN = 0x02;
    public static final byte CMD_SCAN_RESULT = 0x03;
    public static final byte WLAN_0 = 0x00;
    public static final byte WLAN_1 = 0x01;
    private LocalSocket client = null;
    private OutputStream os = null;
    private InputStream is = null;
    private int timeout = 20000;

    public void connect() {
        try {
            client = new LocalSocket();
            if (!client.isConnected()) {
                client.connect(new LocalSocketAddress(ADDRESS, LocalSocketAddress.Namespace.ABSTRACT));
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public void send(byte[] data) {
        try {
            os = client.getOutputStream();
            os.write(data);
            os.flush();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public String recv() {
        String result = null;
        try {
            is = client.getInputStream();
            byte[] bytes = new byte[1024];
            int read = is.read(bytes, 0, 1024);
            if (read < 0) {
                return result;
            }
            result = new String(bytes, 0, read);
        } catch (IOException e) {
            e.printStackTrace();
        }
        return result;
    }

    public void close()
        try {
            if (os != null) {
                os.close();
            }
            if (is != null) {
                is.close();
            }
            if (client != null) {
                if (client.isConnected()) {
                    client.close();
                    client = null;
                } else {
                    client = null;
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
